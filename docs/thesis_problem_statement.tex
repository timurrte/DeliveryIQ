\section{Математична постановка задачі}\label{sec:formulation}

\subsection{Формалізація транспортної мережі}

Транспортну мережу формалізуємо як \textbf{мультимодальний зважений орієнтований граф} \cite{toth2014}:
\[
    G = (V,\; E,\; \Phi,\; \Psi),
\]
де:
\begin{itemize}[leftmargin=2em]
    \item $V = \{v_1, v_2, \ldots, v_n\}$ --- скінченна множина вузлів мережі;
    \item $E \subseteq V \times V$ --- множина орієнтованих ребер (доріг);
    \item $\Phi : V \to \vset{A}$ --- функція атрибутів вузлів;
    \item $\Psi : E \to \vset{B}$ --- функція атрибутів ребер.
\end{itemize}

\begin{definition}
\textbf{Вузол} $v_i \in V$ характеризується кортежем атрибутів:
\[
    \Phi(v_i) = \bigl(\mathrm{id}_i,\; \lambda_i,\; \mu_i,\; \delta_i^{\min},\; \delta_i^{\max},\; s_i\bigr),
\]
де $\mathrm{id}_i$ --- унікальний ідентифікатор; $(\lambda_i, \mu_i) \in \Real^2$ --- географічні координати (широта, довгота); $[\delta_i^{\min}, \delta_i^{\max}]$ --- часове вікно роботи хабу (наприклад, 08:00--21:00); $s_i \geq 0$ --- час обслуговування у вузлі (розвантаження, оформлення тощо).
\end{definition}

\begin{definition}
\textbf{Ребро} $e_{ij} = (v_i, v_j) \in E$ характеризується кортежем:
\[
    \Psi(e_{ij}) = \bigl(\ell_{ij},\; M_{ij}\bigr),
\]
де $\ell_{ij} \in \Real_{>0}$ --- довжина сегменту (метри); $M_{ij} \subseteq \mathcal{T}$ --- множина допустимих для цього ребра типів транспортних засобів.
\end{definition}

\subsection{Типи транспортних засобів та мультимодальність}

Визначимо множину типів транспортних засобів:
\[
    \mathcal{T} = \{\mathrm{Car},\; \mathrm{Bike},\; \mathrm{Walk}\},
\]
та визначимо підмножину ребер, доступних для кожного типу транспорту:
\[
    E_\theta = \bigl\{\, e_{ij} \in E \mid \theta \in M_{ij} \,\bigr\}, \qquad \theta \in \mathcal{T}.
\]
Підмножини $E_\theta$ можуть перетинатися: одне ребро може бути доступним для кількох типів транспорту.

Для кожного типу транспорту $\theta \in \mathcal{T}$ введемо характеристичну швидкість $\bar{u}_\theta$ (м/с) та вантажну місткість одиниці $Q_\theta$ (кг):
\[
    \bar{u}_{\mathrm{Car}} > \bar{u}_{\mathrm{Bike}} > \bar{u}_{\mathrm{Walk}}, \qquad
    Q_{\mathrm{Car}} \geq Q_{\mathrm{Bike}} \geq Q_{\mathrm{Walk}}.
\]

\textbf{Парк транспортних засобів.} Визначимо скінченну множину наявних транспортних засобів:
\[
    \mathcal{V} = \{w_1, w_2, \ldots, w_F\}, \qquad F = |\mathcal{V}|,
\]
де кожному транспортному засобу $w \in \mathcal{V}$ відповідає тип $\theta_w \in \mathcal{T}$. Кількість одиниць кожного типу:
\[
    F_\theta = \bigl|\{\, w \in \mathcal{V} \mid \theta_w = \theta \,\}\bigr|, \qquad \theta \in \mathcal{T}.
\]
Вантажна місткість транспортного засобу $w$ дорівнює $Q_{\theta_w}$.

\subsection{Формалізація задачі маршрутизації}

Нехай задано \textbf{депо} $v_0 \in V$ та множину пакунків $\mathcal{K} = \{k_1, k_2, \ldots, k_m\}$. Кожен пакунок $k \in \mathcal{K}$ характеризується парою:
\[
    k = \bigl(d_k,\; q_k\bigr),
\]
де $d_k \in V$ --- вузол призначення; $q_k \in \Real_{>0}$ --- маса пакунку. Усі пакунки відправляються з депо $v_0$.

Реальна адреса доставки може не збігатися з жодним вузлом графу $G$. У такому разі $d_k$ визначається як \textbf{найближчий вузол} графу до фактичних координат адреси:
\[
    d_k = \arg\min_{v \in V} \bigl\| (\lambda_v,\, \mu_v) - (\lambda_k^{\mathrm{addr}},\, \mu_k^{\mathrm{addr}}) \bigr\|,
\]
де $(\lambda_k^{\mathrm{addr}}, \mu_k^{\mathrm{addr}})$ --- геокодовані координати адреси. Залишкова відстань від $d_k$ до фактичної адреси долається пішки і враховується в механізмі «останньої милі» (\cite{wang2014}).

\subsubsection{Агрегований граф доставки}

Для формулювання задачі маршрутизації введемо \textbf{агрегований граф доставки} над множиною вузлів:
\[
    N = \{0,\, 1,\, 2,\, \ldots,\, m\},
\]
де вузол $0$ відповідає депо $v_0$, а вузол $i$ ($i = 1, \ldots, m$) --- пункту призначення пакунку $k_i$ (вузлу $d_{k_i} \in V$).

Для кожної пари вузлів $i, j \in N$ ($i \neq j$) та транспортного засобу $w \in \mathcal{V}$ визначимо \textbf{час переїзду} $t_{ij}^{(w)}$, що враховує механізм «останньої милі». Нехай $P_{ij} = (e_1, e_2, \ldots, e_{r_{ij}})$ --- найкоротший шлях від вузла $i$ до вузла $j$ у вуличному графі $G$ з $r_{ij}$ ребрами.

\textbf{Індекс розбиття} $s_{ij}^{(w)}$ визначається як максимальний префікс шляху, прохідний для типу $\theta_w$:
\[
    s_{ij}^{(w)} = \max\Bigl\{\, l \in \{0, \ldots, r_{ij}\} \;\Big|\; \forall\, l' \in \{1, \ldots, l\}:\; \theta_w \in M_{e_{l'}} \,\Bigr\}.
\]
Тобто транспортний засіб $w$ проїжджає ребра $e_1, \ldots, e_{s_{ij}^{(w)}}$ (основна частина), а ребра $e_{s_{ij}^{(w)}+1}, \ldots, e_{r_{ij}}$ долаються пішки («остання миля»). Зокрема, $s_{ij}^{(w)}=0$  означає, що жодне ребро не прохідне для $\theta_{w}$ і весь шлях долається пішки. Час переїзду:
\[
    t_{ij}^{(w)} = \underbrace{\sum_{l=1}^{s_{ij}^{(w)}} \frac{\ell_{e_l}}{\bar{u}_{\theta_w}}}_{\text{основна частина}} \;+\; \underbrace{\sum_{l=s_{ij}^{(w)}+1}^{r_{ij}} \frac{\ell_{e_l}}{\bar{u}_{\mathrm{Walk}}}}_{\text{«остання миля»}}.
\]

Позначимо \textbf{довжину «останньої милі»}:
\[
    \Delta_{ij}^{(w)} = \sum_{l=s_{ij}^{(w)}+1}^{r_{ij}} \ell_{e_l}.
\]

Шлях $P_{ij}$ вважається \textbf{здійсненним} для засобу $w$, якщо виконуються обидві умови:
\begin{enumerate}[leftmargin=2em, label=(\alph*)]
    \item кожне ребро «останньої милі» прохідне пішки: $\mathrm{Walk} \in M_{e_l}$ для $l > s_{ij}^{(w)}$;
    \item довжина «останньої милі» не перевищує порогу: $\Delta_{ij}^{(w)} \leq \Delta_{\max}$.
\end{enumerate}
Якщо хоча б одна з умов порушена, покладемо $t_{ij}^{(w)} = +\infty$. Параметр $\Delta_{\max}$ (типове значення --- 100~м) обмежує відстань, яку кур'єр може пронести вантаж пішки.

\subsubsection{Змінні рішення}

Введемо такі змінні:

\begin{enumerate}[leftmargin=2em]
    \item \textbf{Маршрутна змінна:}
    \[
        x_{ijw} \in \{0, 1\}, \quad i, j \in N,\; i \neq j,\; w \in \mathcal{V},
    \]
    де $x_{ijw} = 1$, якщо транспортний засіб $w$ переїжджає безпосередньо з вузла $i$ до вузла $j$.

    \item \textbf{Індикатор використання ТЗ:}
    \[
        y_w \in \{0, 1\}, \quad w \in \mathcal{V},
    \]
    де $y_w = 1$, якщо засіб $w$ задіяний у доставці.

    \item \textbf{Час прибуття:}
    \[
        \tau_i^{(w)} \in \Real_{\geq 0}, \quad i \in N,\; w \in \mathcal{V}
    \]
    --- момент часу прибуття засобу $w$ до вузла $i$ (допоміжна змінна для елімінації підциклів та контролю часових вікон).
\end{enumerate}

\subsection{Цільова функція та обмеження}

\textbf{Задача оптимізації} формулюється як задача змішаного цілочислового програмування:

\[
    \boxed{
    \min_{x,\, y} \; F
    = A \cdot \underbrace{\sum_{w \in \mathcal{V}} \sum_{\substack{i,j \in N \\ i \neq j}} t_{ij}^{(w)}\, x_{ijw}}_{T_{\mathrm{total}}}
    \;+\; B \cdot \underbrace{\sum_{w \in \mathcal{V}} y_w\vphantom{\sum_{\substack{i,j}}}}_{|\mathcal{V}_{\mathrm{act}}|}
    }
\]
де $A, B \geq 0$ --- вагові коефіцієнти ($A \gg B$): пріоритетом є мінімізація сумарного часу доставки $T_{\mathrm{total}}$, а вторинним критерієм --- мінімізація кількості задіяних ТЗ -- $|\mathcal{V}_{\mathrm{act}}|$.

\medskip

За умови виконання таких обмежень:

\begin{enumerate}[label=\textbf{О\arabic*.}, leftmargin=3em]
    \item \textbf{(Повнота доставки):} кожен пункт призначення відвідується рівно одним ТЗ:
    \[
        \sum_{w \in \mathcal{V}} \sum_{\substack{j \in N \\ j \neq i}} x_{ijw} = 1, \quad \forall\, i \in N \setminus \{0\}.
    \]

    \item \textbf{(Збереження потоку):} якщо ТЗ $w$ прибуває до вузла, він із нього і від'їжджає:
    \[
        \sum_{\substack{j \in N \\ j \neq i}} x_{jiw} = \sum_{\substack{j \in N \\ j \neq i}} x_{ijw}, \quad \forall\, i \in N,\; \forall\, w \in \mathcal{V}.
    \]

    \item \textbf{(Зв'язок із депо):} ТЗ виїжджає з депо тоді й лише тоді, коли він задіяний:
    \[
        \sum_{j \in N \setminus \{0\}} x_{0jw} = y_w, \qquad
        \sum_{j \in N \setminus \{0\}} x_{j0w} = y_w, \quad \forall\, w \in \mathcal{V}.
    \]

    \item \textbf{(Вантажна місткість):} сумарна маса пакунків у маршруті ТЗ $w$ не перевищує його місткість:
    \[
        \sum_{i \in N \setminus \{0\}} q_{k_i} \cdot z_{iw} \leq Q_{\theta_w}, \quad \forall\, w \in \mathcal{V},
    \]
    де $z_{iw} = \displaystyle\sum_{\substack{j \in N \\ j \neq i}} x_{ijw} \in \{0, 1\}$ --- індикатор того, що вузол $i$ обслуговується засобом $w$; $k_{i}$ --- пакунок, що відповідає вузлу $i \in N \setminus \{0\}$.

    \item \textbf{(Часові вікна та елімінація підциклів):} для кожного переїзду час прибуття узгоджується з часом проходження:
    \[
        \tau_j^{(w)} \geq \tau_i^{(w)} + s_i + t_{ij}^{(w)} - \mathcal{M}\bigl(1 - x_{ijw}\bigr), \quad \forall\, i,j \in N,\; i \neq j,\; \forall\, w \in \mathcal{V},
    \]
    де $s_i \geq 0$ --- час обслуговування у вузлі $i$ (визначений у кортежі $\Phi(v_i)$); $\mathcal{M} > 0$ --- достатньо велика константа. Час прибуття до кожного пункту має потрапляти у вікно роботи хабу:
    \[
        \delta_i^{\min} \leq \tau_i^{(w)} \leq \delta_i^{\max}, \quad \forall\, i \in N \setminus \{0\},\; \forall\, w \in \mathcal{V}.
    \]

    Це одночасно виконує дві функції: (1)~елімінація підциклів за схемою Міллера--Такера--Землін (MTZ) \cite{mtz1960}, оскільки змінні $\tau_i^{(w)}$ зростають вздовж маршруту; (2)~контроль часових вікон роботи хабів.

    \item \textbf{(Допустимість маршруту):} ТЗ $w$ може використати дугу $(i, j)$ лише якщо шлях у графі $G$ є здійсненним (основна частина прохідна для $\theta_w$, «остання миля» --- пішки):
    \[
        x_{ijw} = 1 \;\Rightarrow\; t_{ij}^{(w)} < +\infty, \quad \forall\, i,j \in N,\; i \neq j,\; \forall\, w \in \mathcal{V}.
    \]
\end{enumerate}

\section{Метод розв'язання: евристика Соломона}\label{sec:solomon}

Задача CVRPTW, сформульована в розділі~\ref{sec:formulation}, належить до класу NP-складних задач комбінаторної оптимізації. Для її наближеного розв'язання застосовано \textbf{евристику послідовних вставок Соломона} (Solomon Insertion Heuristic~I1)~ \cite{solomon1987} --- конструктивний жадібний алгоритм, спеціально розроблений для задач маршрутизації з часовими вікнами (VRPTW) \cite{braysy2005}.

На відміну від метаевристик (генетичні алгоритми, імітація відпалу тощо), евристика Соломона будує допустимий розв'язок за \textbf{один прохід}, послідовно вставляючи необслуговані пункти в поточні маршрути. Це гарантує поліноміальний час роботи та допустимість кожного проміжного розв'язку за обмеженнями місткості й часових вікон.

\subsection{Представлення розв'язку}

Розв'язок задачі складається з множини маршрутів $\{R_1, R_2, \ldots, R_K\}$, де $K$~--- кількість задіяних транспортних засобів ($K \leq F$). Маршрут $R_r$ для $r$-го транспортного засобу $w_r$ є впорядкованою послідовністю пунктів призначення:
\[
    R_r = \bigl(0,\; i_1,\; i_2,\; \ldots,\; i_{n_r},\; 0\bigr), \qquad i_l \in N \setminus \{0\},
\]
де $n_r = |R_r| - 2$~--- кількість пунктів призначення в маршруті, а вузол $0$ (депо) є початком і кінцем маршруту.

Для кожного маршруту підтримуються:
\begin{itemize}[leftmargin=2em]
    \item \textbf{поточне завантаження:} $L_r = \displaystyle\sum_{l=1}^{n_r} q_{k_{i_l}}$~--- сумарна маса пакунків;
    \item \textbf{час прибуття:} $\tau_{i_l}^{(w_r)}$~--- момент прибуття транспортного засобу $w_r$ до $l$-го пункту призначення маршруту. Введемо \textbf{фактичний час початку обслуговування}:
    \[
        b_{i_l} = \max\!\bigl\{\tau_{i_l}^{(w_r)},\; \delta_{i_l}^{\min}\bigr\},
    \]
    тобто якщо ТЗ прибуває раніше за відкриття часового вікна ($\tau_{i_l}^{(w_r)} < \delta_{i_l}^{\min}$), він очікує до моменту $\delta_{i_l}^{\min}$. Час прибуття до наступного пункту обчислюється рекурентно:
    \[
        \tau_{i_{l+1}}^{(w_r)} = b_{i_l} + s_{i_l} + t_{i_l,\, i_{l+1}}^{(w_r)},
    \]
    де $s_{i_l} \geq 0$~--- час обслуговування у вузлі $i_l$ (визначений у кортежі $\Phi(v_{i_l})$); $\tau_0^{(w_r)} = \delta_0^{\min}$~--- час відправлення з депо. Допустимість вставки вимагає $\tau_{i_l}^{(w_r)} \leq \delta_{i_l}^{\max}$ для кожного пункту $i_l$ маршруту (включно з поверненням до депо).
\end{itemize}

\subsection{Критерій допустимості вставки}

Нехай поточний маршрут $R_r = (0, i_1, \ldots, i_p, \ldots, i_{n_r}, 0)$ обслуговується транспортним засобом $w_r$ і розглядається вставка необслугованого пункту $u \in \mathcal{U}$ між позиціями $i_p$ та $i_{p+1}$ (де $i_0 = i_{n_r+1} = 0$). Вставка є \textbf{допустимою}, якщо одночасно виконуються три умови:

\medskip

\textbf{Д1. Допустимість маршруту в графі} (відповідає обмеженню~О6)\textbf{.} Обидва переїзди мають скінченний час:
\[
    t_{i_p,\, u}^{(w_r)} < +\infty \quad \text{та} \quad t_{u,\, i_{p+1}}^{(w_r)} < +\infty.
\]

\textbf{Д2. Обмеження вантажної місткості.} Додавання пакунку $k_u$ не перевищує місткість ТЗ:
\[
    L_r + q_{k_u} \leq Q_{\theta_{w_r}}.
\]

\textbf{Д3. Обмеження часових вікон.} Час прибуття до пункту $u$ та до \textbf{всіх наступних} пунктів маршруту залишається в межах відповідних часових вікон. Новий час прибуття до $u$:
\[
    \tau_u^{\mathrm{new}} = b_{i_p} + s_{i_p} + t_{i_p,\, u}^{(w_r)},
\]
де $b_{i_p} = \max\!\bigl\{\tau_{i_p}^{(w_r)},\; \delta_{i_p}^{\min}\bigr\}$~--- фактичний час початку обслуговування у вузлі $i_p$.
Необхідно, щоб $\delta_u^{\min} \leq \tau_u^{\mathrm{new}} \leq \delta_u^{\max}$. Крім того, вставка $u$ зсуває час прибуття до всіх пунктів $i_{p+1}, i_{p+2}, \ldots, i_{n_r}$ та повернення до депо. Позначимо \textbf{зсув часу} (push-forward):
\[
    \mathrm{PF}_u = \max\!\Bigl\{0,\;\bigl(b_u^{\mathrm{new}} + s_u + t_{u,\, i_{p+1}}^{(w_r)}\bigr) - \tau_{i_{p+1}}^{(w_r)}\Bigr\},
\]
де $b_u^{\mathrm{new}} = \max\!\bigl\{\tau_u^{\mathrm{new}},\; \delta_u^{\min}\bigr\}$~--- фактичний час початку обслуговування у вставленому вузлі $u$. Оператор $\max\{0, \cdot\}$ є принциповим для VRPTW: якщо транспортний засіб прибуває до $i_{p+1}$ раніше за $\delta_{i_{p+1}}^{\min}$, час очікування «поглинає» зсув, і $\mathrm{PF}_u$ не поширюється далі \cite{solomon1987}. Зсув каскадно поширюється вздовж маршруту: для кожного наступного пункту $i_l$ ($l = p+1, \ldots, n_r$) та для повернення до депо оновлений час прибуття $\tau_{i_l}^{\mathrm{new}} = \tau_{i_l}^{(w_r)} + \mathrm{PF}_u$ має задовольняти $\tau_{i_l}^{\mathrm{new}} \leq \delta_{i_l}^{\max}$.

\subsection{Критерії вставки (Solomon I1)}

Евристика I1 використовує два критерії для вибору \textit{найкращої позиції} та \textit{найкращого пункту} для вставки.

\subsubsection{Критерій $c_1$: вартість вставки у позицію}

Для кожної допустимої позиції $(i_p, i_{p+1})$ у маршруті $R_r$ та кандидата $u$ обчислюється \textbf{вартість вставки}~--- зважена комбінація додаткового часу переїзду та зсуву розкладу:

\[
    c_{11}(i_p, u, i_{p+1}) = t_{i_p,\, u}^{(w_r)} + t_{u,\, i_{p+1}}^{(w_r)} - \mu_1 \cdot t_{i_p,\, i_{p+1}}^{(w_r)},
\]
де $\mu_1 \geq 0$~--- параметр (типово $\mu_1 = 1$). Доданок $c_{11}$ вимірює \textbf{додатковий час переїзду}, що виникає при вставці $u$: час нового шляху $i_p \to u \to i_{p+1}$ мінус час прямого шляху $i_p \to i_{p+1}$.

\[
    c_{12}(i_p, u, i_{p+1}) = \tau_{i_{p+1}}^{\mathrm{new}} - \tau_{i_{p+1}}^{(w_r)} = \mathrm{PF}_u,
\]
де $c_{12}$~--- \textbf{зсув часу} (push-forward) для наступного пункту маршруту після вставки $u$.

\textbf{Зважений критерій вставки:}
\[
    c_1(i_p, u, i_{p+1}) = \alpha_1 \cdot c_{11}(i_p, u, i_{p+1}) + \alpha_2 \cdot c_{12}(i_p, u, i_{p+1}),
\]
де $\alpha_1, \alpha_2 \geq 0$, $\alpha_1 + \alpha_2 = 1$~--- вагові коефіцієнти. При $\alpha_1 = 1, \alpha_2 = 0$ критерій враховує лише додатковий час переїзду; при $\alpha_1 = 0, \alpha_2 = 1$~--- лише зсув розкладу.

\textbf{Найкраща позиція} для вставки пункту $u$ в маршрут $R_r$:
\[
    c_1^{*}(u, R_r) = \min_{p \in \{0, \ldots, n_r\}} c_1(i_p, u, i_{p+1}),
\]
де мінімум береться лише по \textit{допустимих} позиціях (що задовольняють умови Д1--Д3).

\subsubsection{Критерій $c_2$: вибір найкращого кандидата}

Серед усіх необслугованих пунктів $u \in \mathcal{U}$, для яких існує хоча б одна допустима позиція в маршруті $R_r$, обирається пункт із \textbf{найбільшою вигодою вставки}:

\[
    c_2(u, R_r) = \lambda \cdot t_{0,\, u}^{(w_r)} - c_1^{*}(u, R_r),
\]
де $\lambda \geq 0$~--- параметр (типово $\lambda = 1$). Перший доданок $\lambda \cdot t_{0,u}^{(w_r)}$ стимулює вставку \textbf{віддалених} від депо пунктів (їх обслуговування окремим маршрутом було б дорожчим); другий доданок штрафує дорогі вставки.

\textbf{Пункт для вставки} обирається як:
\[
    u^{*} = \arg\max_{u \in \mathcal{U}} c_2(u, R_r).
\]
Максимізація $c_2$ забезпечує жадібний вибір пункту, вставка якого дає найбільший виграш відносно альтернативи прямого обслуговування з депо.

\subsection{Ініціалізація маршрутів (вибір seed-клієнта)}

При створенні нового маршруту $R_r$ необхідно обрати \textbf{початковий пункт} (seed) з множини необслугованих пунктів $\mathcal{U}$. Застосовується критерій \textbf{найвіддаленішого пункту}:
\[
    u_{\mathrm{seed}} = \arg\max_{u \in \mathcal{U}} t_{0,\, u}^{(w_r)},
\]
тобто обирається пункт із найбільшим часом переїзду від депо. Це мотивовано тим, що віддалені пункти складніше «вмістити» в існуючі маршрути на пізніших ітераціях, тому їх доцільно обслуговувати першими.

Початковий маршрут: $R_r = (0,\; u_{\mathrm{seed}},\; 0)$, із завантаженням $L_r = q_{k_{u_{\mathrm{seed}}}}$.

\subsection{Основний алгоритм}

Евристика Соломона I1 будує розв'язок послідовно, маршрут за маршрутом. Нехай $\mathcal{U}$~--- множина ще не обслугованих пунктів (ініціалізується як $\mathcal{U} = N \setminus \{0\}$).

\begin{enumerate}[leftmargin=2em]
    \item \textbf{Ініціалізація:} $\mathcal{U} \leftarrow N \setminus \{0\}$; $K \leftarrow 0$ (кількість маршрутів).

    \item \textbf{Поки $\mathcal{U} \neq \varnothing$:}
    \begin{enumerate}[label=(\alph*)]
        \item \textbf{Новий маршрут:} обрати ТЗ $w_r$ з парку, $K \leftarrow K + 1$.
        \item \textbf{Seed:} $u_{\mathrm{seed}} \leftarrow \arg\max_{u \in \mathcal{U}} t_{0,u}^{(w_r)}$; \; $R_r \leftarrow (0,\; u_{\mathrm{seed}},\; 0)$; \; $\mathcal{U} \leftarrow \mathcal{U} \setminus \{u_{\mathrm{seed}}\}$.
        \item \textbf{Послідовна вставка.} Повторювати:
        \begin{enumerate}[label=\roman*.]
            \item Для кожного $u \in \mathcal{U}$ обчислити $c_1^{*}(u, R_r)$ по всіх допустимих позиціях.
            \item Визначити $u^{*} = \arg\max_{u \in \mathcal{U}} c_2(u, R_r)$ серед пунктів, для яких існує допустима позиція.
            \item Якщо $u^{*}$ знайдено: вставити $u^{*}$ у найкращу позицію маршруту $R_r$; \; $\mathcal{U} \leftarrow \mathcal{U} \setminus \{u^{*}\}$; оновити $L_r$ та $\tau^{(w_r)}$.
            \item Якщо жоден пункт не може бути вставлений допустимо~--- \textbf{закрити} маршрут $R_r$ і перейти до кроку~(a).
        \end{enumerate}
    \end{enumerate}

    \item \textbf{Результат:} множина маршрутів $\{R_1, R_2, \ldots, R_K\}$.
\end{enumerate}

\subsection{Вартість розв'язку}

Після побудови всіх маршрутів загальна вартість розв'язку обчислюється за цільовою функцією, визначеною в розділі~\ref{sec:formulation}:
\[
    F = A \cdot \sum_{r=1}^{K} \sum_{l=0}^{n_r} t_{i_l,\, i_{l+1}}^{(w_r)} \;+\; B \cdot K,
\]
де перший доданок~--- сумарний час переїзду по всіх маршрутах, другий~--- кількість задіяних ТЗ.

Оскільки евристика I1 є конструктивною, кожен побудований маршрут \textbf{гарантовано допустимий} за обмеженнями місткості (Д2) та часових вікон (Д3). Штрафні функції, необхідні в метаевристиках для обробки порушень обмежень, тут не потрібні~--- допустимість забезпечується на етапі перевірки кожної вставки.

\subsection{Обчислювальна складність}

Позначимо $m = |N \setminus \{0\}|$~--- загальну кількість пунктів призначення.

На кожній ітерації вставки алгоритм перебирає $O(m)$ кандидатів і для кожного~--- $O(m)$ позицій у поточному маршруті (перевірка допустимості за Д1--Д3 виконується за $O(m)$ через поширення зсуву PF). Усього виконується $m$ вставок (кожен пункт вставляється рівно один раз). Отже, загальна складність при фіксованому типі ТЗ:
\[
    O(m^3).
\]
У мультимодальному варіанті з $|\mathcal{T}|$ типами транспортних засобів обчислення матриці часів переїзду виконується для кожного типу окремо, що дає складність $O(|\mathcal{T}| \cdot m^3)$. Оскільки $|\mathcal{T}|$ є константою ($|\mathcal{T}| = 3$ у даній роботі), асимптотична складність залишається $O(m^3)$.

На практиці середня довжина маршруту $\bar{n} \ll m$, і типова складність наближається до $O(m^2 \cdot \bar{n})$.

Для задач із десятками та сотнями пунктів призначення це забезпечує час роботи порядку мілісекунд, що суттєво швидше за ітеративні метаевристики.

\subsection{Параметри алгоритму}

Евристика I1 контролюється п'ятьма параметрами:

\begin{center}
\begin{tabular}{@{} l c p{7.5cm} @{}}
\toprule
\textbf{Параметр} & \textbf{Типове значення} & \textbf{Призначення} \\
\midrule
$\alpha_1$ & $1{,}0$ & вага додаткового часу переїзду в $c_1$ \\
$\alpha_2$ & $0{,}0$ & вага зсуву розкладу в $c_1$ \\
$\mu_1$ & $1{,}0$ & коефіцієнт вирахування прямого шляху в $c_{11}$ \\
$\lambda$ & $1{,}0$ & вага відстані від депо в $c_2$ \\
$\Delta_{\max}$ & $100$\,м & максимальна довжина «останньої милі» \\
\bottomrule
\end{tabular}
\end{center}

При $\alpha_1 = 1, \alpha_2 = 0, \lambda = 1$ евристика мінімізує додатковий час переїзду при кожній вставці та віддає пріоритет віддаленим пунктам~--- це відповідає класичній конфігурації Solomon~I1 для задач VRPTW із географічно розподіленими клієнтами.

\begin{remark}
Евристика Соломона є конструктивною евристикою і не гарантує знаходження глобального оптимуму. Проте вона будує допустимі розв'язки за поліноміальний час $O(m^3)$, що суттєво краще за точні методи (повний перебір зі складністю $O(m!)$).
\end{remark}
